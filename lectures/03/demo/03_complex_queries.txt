    "## Window Functions\n",
    "\n",
    "Window functions are powerful tools for complex analytics. They allow us to perform calculations across sets of rows related to the current row:"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "source": [
    "%%sql\n",
    "-- Rank participants by BMI within age groups\n",
    "-- This shows how to use PARTITION BY to create groups\n",
    "SELECT \n",
    "    d.seqn,\n",
    "    d.age,\n",
    "    e.bmxbmi,\n",
    "    RANK() OVER (\n",
    "        PARTITION BY \n",
    "            CASE \n",
    "                WHEN age < 30 THEN '18-29'\n",
    "                WHEN age < 40 THEN '30-39'\n",
    "                WHEN age < 50 THEN '40-49'\n",
    "                ELSE '50+'\n",
    "            END\n",
    "        ORDER BY e.bmxbmi DESC\n",
    "    ) AS bmi_rank\n",
    "FROM demographics d\n",
    "JOIN examination e ON d.seqn = e.seqn\n",
    "LIMIT 10;"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "source": [
    "%%sql\n",
    "-- Calculate running average of lab values\n",
    "-- This shows how to use window frames for moving calculations\n",
    "SELECT \n",
    "    seqn,\n",
    "    lbxglu,\n",
    "    AVG(lbxglu) OVER (\n",
    "        ORDER BY seqn \n",
    "        ROWS BETWEEN 2 PRECEDING AND 2 FOLLOWING\n",
    "    ) AS running_avg\n",
    "FROM laboratory\n",
    "LIMIT 10;"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "source": [
    "%%sql\n",
    "-- Compare each participant's BMI to their age group average\n",
    "-- This shows how to use window functions for comparative analysis\n",
    "WITH age_groups AS (\n",
    "    SELECT \n",
    "        d.seqn,\n",
    "        d.age,\n",
    "        e.bmxbmi,\n",
    "        CASE \n",
    "            WHEN age < 30 THEN '18-29'\n",
    "            WHEN age < 40 THEN '30-39'\n",
    "            WHEN age < 50 THEN '40-49'\n",
    "            ELSE '50+'\n",
    "        END AS age_group\n",
    "    FROM demographics d\n",
    "    JOIN examination e ON d.seqn = e.seqn\n",
    ")\n",
    "SELECT \n",
    "    seqn,\n",
    "    age,\n",
    "    bmxbmi,\n",
    "    age_group,\n",
    "    AVG(bmxbmi) OVER (PARTITION BY age_group) AS group_avg,\n",
    "    bmxbmi - AVG(bmxbmi) OVER (PARTITION BY age_group) AS diff_from_avg\n",
    "FROM age_groups\n",
    "LIMIT 10;"
   ]
